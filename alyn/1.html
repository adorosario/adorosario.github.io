#/usr/bin/perl
#
use strict;
use warnings;
use Data::Dumper;
use POSIX;

open(FILES, "find -type f -print |") || die "Find Command Failed"; my $num = 0;
while(<FILES>) { 

        chomp;
        my $filename="$_";
        if ( $filename =~ m/^\.\/(\d+) .*?/ ) { 
          $num = "$1";
          print "$1\t$filename\n"; 
        }
        system("txt2html --outfile $num.html '$filename'");

}
close(FILES);
